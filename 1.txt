function Perform-RandomCalculations {
    $stopTime = Get-Random -Minimum 60 -Maximum 101
    $endTime = (Get-Date).AddSeconds($stopTime)

    while ((Get-Date) -lt $endTime) {
        $dummyCalculation = [Math]::Sin((Get-Random) * 1000)
    }
}

Start-Job -ScriptBlock { Perform-RandomCalculations }
